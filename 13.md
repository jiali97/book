### 13 Roman to Integer

> 纯模拟

```python
class Solution(object):
    def romanToInt(self, s):
        """
        :type s: str
        :rtype: int
        """
        d = {   "I":1,
                "IV":4,
                "V":5,
                "IX":9,
                "X":10,
                "XL":40,
                "L":50,
                "XC":90,
                "C":100,
                "CD":400,
                "D":500,
                "CM":900,
                "M":1000 }
        ans = 0
        l = 0
        while l<len(s):
            tmp = s[l:l+2]
            if tmp in ["IV","IX","XL","XC","CD","CM"]:
                l+=2
            else:
                tmp = s[l:l+1]
                l+=1
            ans += d[tmp]
        return ans
```
